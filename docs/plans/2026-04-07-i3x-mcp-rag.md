```typescript
/** Minimal ObjectTree interface that I3xRag depends on. */
function createMockObjectTree(objects: I3xObject[], children: Map<string, Set<string>>) {
    const objectMap = new Map(objects.map(o => [o.elementId, o]));
    return {
        getObjects: () => Array.from(objectMap.values()),
        getObject: (id: string) => objectMap.get(id),
        getObjectTypes: () => [],
        getChildElementIds: (id: string) => Array.from(children.get(id) ?? []),
        getDescendantLeafIds: (id: string, maxDepth = 0, depth = 0): string[] => {
            if (maxDepth > 0 && depth >= maxDepth) return [];
            const childIds = Array.from(children.get(id) ?? []);
            const leaves: string[] = [];
            for (const cid of childIds) {
                const c = objectMap.get(cid);
                if (c && !c.isComposition) leaves.push(cid);
                else leaves.push(...createMockObjectTree(objects, children).getDescendantLeafIds(cid, maxDepth, depth + 1));
            }
            return leaves;
        },
        getRelated: (id: string) => {
            const obj = objectMap.get(id);
            if (!obj) return [];
            const result: I3xObject[] = [];
            if (obj.parentId && obj.parentId !== "/") {
                const parent = objectMap.get(obj.parentId);
                if (parent) result.push(parent);
            }
            const childIds = children.get(id);
            if (childIds) for (const cid of childIds) {
                const child = objectMap.get(cid);
                if (child) result.push(child);
            }
            return result;
        },
    };
}

function createMockValueCache(values: Map<string, I3xVqt>) {

        getValue: (id: string) => {
            const vqt = values.get(id);
            if (!vqt) return null;
            return { elementId: id, isComposition: false, ...vqt };
```

```typescript
// src/rag/i3x-rag.ts
import Graph from "graphology";
import MiniSearch from "minisearch";
import type { I3xObject, I3xVqt } from "../types/i3x.js";

/** Minimal interface for what I3xRag needs from ObjectTree. */
export interface ObjectTreeLike {
    getObjects(opts?: any): I3xObject[];
    getObject(elementId: string): I3xObject | undefined;
    getObjectTypes(): Array<{ elementId: string; displayName: string }>;
    getChildElementIds(elementId: string): string[];
    getDescendantLeafIds(elementId: string, maxDepth?: number): string[];
    getRelated(elementId: string): I3xObject[];
}

/** Minimal interface for what I3xRag needs from ValueCache. */
export interface ValueCacheLike {
    getValue(elementId: string): { elementId: string; isComposition: boolean; value: unknown; quality: string; timestamp: string } | null;
}

/** Minimal interface for what I3xRag needs from History. */
export interface HistoryLike {
    queryHistory(elementId: string, startTime: string, endTime: string): Promise<I3xVqt[]>;
}

interface I3xRagOpts {
    objectTree: ObjectTreeLike;
    valueCache: ValueCacheLike;
    history: HistoryLike;
}

export class I3xRag {
    private objectTree: ObjectTreeLike;
    private valueCache: ValueCacheLike;
    private history: HistoryLike;
    private graph: Graph;
    private searchIndex: MiniSearch;

    constructor(opts: I3xRagOpts) {
        this.objectTree = opts.objectTree;
        this.valueCache = opts.valueCache;
        this.history = opts.history;
        this.graph = new Graph({ type: "undirected" });
        this.searchIndex = new MiniSearch({
            fields: ["displayName", "typeElementId"],
            storeFields: ["displayName", "typeElementId", "parentId", "isComposition"],
            idField: "elementId",
        });
    }

    init(): void {
        this.buildGraph();
        this.buildSearchIndex();
    }

    rebuild(): void {
        this.graph.clear();
        this.searchIndex.removeAll();
        this.buildGraph();
        this.buildSearchIndex();
    }

    nodeCount(): number {
        return this.graph.order;
    }

    edgeCount(): number {
        return this.graph.size;
    }

    private buildGraph(): void {
        const objects = this.objectTree.getObjects();
        for (const obj of objects) {
            this.graph.addNode(obj.elementId, {
                displayName: obj.displayName,
                typeElementId: obj.typeElementId,
                isComposition: obj.isComposition,
                parentId: obj.parentId,
            });
        }
        for (const obj of objects) {
            if (obj.parentId && obj.parentId !== "/") {
                if (this.graph.hasNode(obj.parentId)) {
                    this.graph.addEdge(obj.parentId, obj.elementId, {
                        label: "parent-child",
                    });
                }
            }
        }
    }

    private buildSearchIndex(): void {
        const objects = this.objectTree.getObjects();
        const docs = objects.map(obj => ({
            elementId: obj.elementId,
            displayName: obj.displayName,
            typeElementId: obj.typeElementId,
            parentId: obj.parentId,
            isComposition: obj.isComposition,
        }));
        this.searchIndex.addAll(docs);
    }
}
```

```typescript
search(query: string, limit: number = 20): SearchResult[] {
    const results = this.searchIndex.search(query, { prefix: true, fuzzy: 0.2 });
    return results.slice(0, limit).map(r => ({
        elementId: r.id as string,
        displayName: r.displayName,
        typeElementId: r.typeElementId,
        score: r.score,
    }));
}

searchByType(typeElementId: string, query: string, limit: number = 20): SearchResult[] {
    const results = this.searchIndex.search(query, {
        prefix: true,
        fuzzy: 0.2,
        filter: (result) => result.typeElementId === typeElementId,
    });
    return results.slice(0, limit).map(r => ({
        elementId: r.id as string,
        displayName: r.displayName,
        typeElementId: r.typeElementId,
        score: r.score,
    }));
}

searchRelated(query: string, hops: number = 1, limit: number = 20): SearchRelatedResult[] {
    const searchResults = this.search(query, limit);
    return searchResults.map(sr => {
        const related = this.neighborhood(sr.elementId, hops);
        return {
            ...sr,
            related: related.filter(n => n.elementId !== sr.elementId),
        };
    });
}
```

```typescript
traverse(elementId: string, hops: number = 1): TraversalNode[] {
    if (!this.graph.hasNode(elementId)) return [];
    const result: TraversalNode[] = [];
    bfsFromNode(this.graph, elementId, (node, attrs, depth) => {
        if (depth > hops) return true; // stop
        if (node !== elementId) {
            result.push({
                elementId: node,
                displayName: attrs.displayName,
                depth,
            });
        }
    });
    return result;
}

findPath(fromId: string, toId: string): string[] | null {
    if (!this.graph.hasNode(fromId) || !this.graph.hasNode(toId)) return null;
    if (fromId === toId) return [fromId];
    const path = bidirectional(this.graph, fromId, toId);
    return path ?? null;
}

// Replace the stub neighborhood() with real implementation:
neighborhood(elementId: string, hops: number = 2): TraversalNode[] {
    return this.traverse(elementId, hops);
}

compositionTree(elementId: string, maxDepth: number = 0): CompositionTreeNode | null {
    const obj = this.objectTree.getObject(elementId);
    if (!obj) return null;
    return this.buildCompositionNode(obj, maxDepth, 0);
}

private buildCompositionNode(obj: I3xObject, maxDepth: number, depth: number): CompositionTreeNode {
    const childIds = this.objectTree.getChildElementIds(obj.elementId);
    let children: CompositionTreeNode[] = [];
    if (maxDepth === 0 || depth < maxDepth) {
        children = childIds
            .map(id => this.objectTree.getObject(id))
            .filter((c): c is I3xObject => c !== undefined)
            .map(c => this.buildCompositionNode(c, maxDepth, depth + 1));
    }
    return {
        elementId: obj.elementId,
        displayName: obj.displayName,
        typeElementId: obj.typeElementId,
        isComposition: obj.isComposition,
        children,
    };
}
```

Methods:

```typescript
relationshipMap(): RelationshipMapEntry[] {
    const counts = new Map<string, number>();
    this.graph.forEachEdge((_edge, _attrs, source, target) => {
        const sourceType = this.graph.getNodeAttribute(source, "typeElementId");
        const targetType = this.graph.getNodeAttribute(target, "typeElementId");
        const key = `${sourceType}->${targetType}`;
        counts.set(key, (counts.get(key) ?? 0) + 1);
    });
    return Array.from(counts.entries()).map(([key, count]) => {
        const [fromType, toType] = key.split("->");
        return { fromType, toType, count };
    });
}

typeSchema(typeElementId: string): TypeSchemaResult | null {
    const instances = this.objectTree.getObjects().filter(o => o.typeElementId === typeElementId);
    if (instances.length === 0) return null;
    const childTypeCounts: Record<string, number> = {};
    for (const inst of instances) {
        const childIds = this.objectTree.getChildElementIds(inst.elementId);
        for (const cid of childIds) {
            const child = this.objectTree.getObject(cid);
            if (child) {
                childTypeCounts[child.typeElementId] = (childTypeCounts[child.typeElementId] ?? 0) + 1;
            }
        }
    }
    return { typeElementId, instanceCount: instances.length, childTypeCounts };
}

valueFilter(opts: {
    quality?: string;
    minValue?: number;
    maxValue?: number;
    missing?: boolean;
}): ValueFilterResult[] {
    const leafObjects = this.objectTree.getObjects().filter(o => !o.isComposition);
    const results: ValueFilterResult[] = [];
    for (const obj of leafObjects) {
        const cached = this.valueCache.getValue(obj.elementId);
        if (opts.missing) {
            if (!cached) {
                results.push({
                    elementId: obj.elementId,
                    displayName: obj.displayName,
                    value: null,
                    quality: "GoodNoData",
                    timestamp: "",
                });
            }
            continue;
        }
        if (!cached) continue;
        if (opts.quality && cached.quality !== opts.quality) continue;
        if (opts.minValue !== undefined && typeof cached.value === "number" && cached.value < opts.minValue) continue;
        if (opts.maxValue !== undefined && typeof cached.value === "number" && cached.value > opts.maxValue) continue;
        results.push({
            elementId: obj.elementId,
            displayName: obj.displayName,
            value: cached.value,
            quality: cached.quality,
            timestamp: cached.timestamp,
        });
    }
    return results;
}

staleValues(thresholdSeconds: number = 300): ValueFilterResult[] {
    const now = Date.now();
    const leafObjects = this.objectTree.getObjects().filter(o => !o.isComposition);
    const results: ValueFilterResult[] = [];
    for (const obj of leafObjects) {
        const cached = this.valueCache.getValue(obj.elementId);
        if (!cached) continue;
        const age = (now - new Date(cached.timestamp).getTime()) / 1000;
        if (age > thresholdSeconds) {
            results.push({
                elementId: obj.elementId,
                displayName: obj.displayName,
                value: cached.value,
                quality: cached.quality,
                timestamp: cached.timestamp,
            });
        }
    }
    return results;
}

async getHistory(elementId: string, startTime: string, endTime: string): Promise<I3xVqt[]> {
    return this.history.queryHistory(elementId, startTime, endTime);
}
```

```typescript
// src/mcp/transport.ts
import type { Application, Request, Response } from "express";
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { StreamableHTTPServerTransport } from "@modelcontextprotocol/sdk/server/streamableHttp.js";

export function mountMcpTransport(app: Application, server: McpServer): void {
    app.post("/mcp", async (req: Request, res: Response) => {
        const transport = new StreamableHTTPServerTransport({
            sessionIdGenerator: undefined,  // stateless
        });
        await server.connect(transport);
        await transport.handleRequest(req, res, req.body);
    });

    // Reject other methods
    app.all("/mcp", (_req: Request, res: Response) => {
        res.status(405).json({ error: "Method not allowed. Use POST." });
    });
}
```
